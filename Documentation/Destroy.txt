"Destroy" - use this function if using pointer type

void Destroy() const

Example:

Bear::List<int*> list;

list.Add(new int(2));
list.Add(new int(100));
list.Add(new int(14));
list.Add(new int(4));
list.Add(new int(2));
list.Add(new int(3));

for (Bear::BearListInt i = 0; i < list.Count(); i++)
{
	std::cout << *list[i] << "\n";
}

list.Destroy();
